theorem neg_neg (a : R) : - -a = a := by
  have h: a + -a = 0 :=by rw[add_neg_cancel]
  exact (eq_neg_of_add_eq_zero h).symm
-- exact (eq_neg_of_add_eq_zero h).symm可以写作rw[←eq_neg_of_add_eq_zero h]，但由于←是rw特有的语法糖，不能写入exact。想要反转，要使用symm函数。

今天写的三个证明，遇到的坑在注释给出：theorem mul_inv_cancel (a : G) : a * a⁻¹ = 1 := by
-- 若希望对于任意变量证明某个引理，需要先声明任意变量；并且在证明时需要先intro
  have h1: ∀b c:G, b⁻¹ * b * c = c := by
    intro b c
    rw[inv_mul_cancel, one_mul]
  have h2: (a*a⁻¹)⁻¹*(a*a⁻¹)*(a*a⁻¹)=(a*a⁻¹) := h1 (a*a⁻¹) (a*a⁻¹)
  have h3: (a*a⁻¹)*(a*a⁻¹)=(a*a⁻¹) := by
    rw[mul_assoc, ←mul_assoc a⁻¹, inv_mul_cancel, one_mul]
-- 若希望对某个已知条件进行rw，需要使用at关键字
  rw[mul_assoc, h3, inv_mul_cancel] at h2
-- 对于exact，使用symm作为←的替代。后者是不规范的，仅在rw里的可用的语法糖
  exact h2.symm

theorem mul_one (a : G) : a * 1 = a := by
  rw[←inv_mul_cancel a, ←mul_assoc, mul_inv_cancel, one_mul]

theorem mul_inv_rev (a b : G) : (a * b)⁻¹ = b⁻¹ * a⁻¹ := by
-- 若使用calc模式，但希望从右向左证明，要先apply Eq.symm
  apply Eq.symm
  calc
  -- 带入变量a * b到定理时，需要使用括号，否则会被认为是先作用在a上，再将作用的结果与b相乘
  -- 如果不使用by + rw，那么就直接使用定理。再次强调，此时需要使用symm而不能使用←
    b⁻¹ * a⁻¹ = b⁻¹ * a⁻¹ * 1 := (mul_one (b⁻¹*a⁻¹)).symm
    _ = b⁻¹ * a⁻¹ * (a * b) * (a * b)⁻¹ :=by rw[←mul_inv_cancel (a * b), ←mul_assoc]
    _ = (a * b)⁻¹ :=by rw[←mul_assoc, mul_assoc b⁻¹, inv_mul_cancel, mul_one, inv_mul_cancel, one_mul]
